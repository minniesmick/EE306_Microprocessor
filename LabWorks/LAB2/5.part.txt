.global _start
_start:
	ldr r0,=LOC
	
	mov r1,#1
	mov r2,#2
	mov r3,#3
	mov r4,#4
	// store words to memory starting from the LOC
	str r1,[r0]
	//store content of r2 to next location
	str r2,[r0,#4 ]//watch r0
	//store r3
	str r3,[r0,#8]
	//store final r4
	str r4,[r0,#12]
	
END:  B  END

LOC: .word 0xA
	 .end
